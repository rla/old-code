/*
Usage:

:$ cat text.txt | rewrite term file.txt > rewritten.txt

Replaces all occurences of <%term%> in input text with the contents
of the file file.txt. 

Raivo Laanemets, 18.11.06
*/

#include <stdio.h>
#include <string.h>
#include <malloc.h>
#include <stdlib.h>

/*
States for the parser.
*/

#define STATE_TEXT 1
#define STATE_TERM 2
#define STATE_TERM_END 3

int main(int argc, char **argv) {
	int c = 0;
	int c1 = 0;
	int state = STATE_TEXT;
	int i = 0;
	int l = 0;	

	if (argc < 3) {
		printf("Usage: rewrite term file.txt\n");
		return 1;
	}

	int t = strlen(argv[1]);
	char *filebuffer;

	FILE *file = fopen(argv[2], "rb");
	if (file == NULL) { return 1; }

	fseek(file, 0, SEEK_END);
	l = ftell(file);
	fseek(file, 0, SEEK_SET);
	filebuffer = malloc(l+1);
	fread(filebuffer, l, 1, file);
	fclose(file);
	filebuffer[l] = '\0';

	char *buffer = malloc(100);

	while((c = getchar()) != EOF) {
		switch (state) {
			case STATE_TEXT: {
				if (c == '<') {
					c1 = getchar();
					if (c1 == EOF) { return 0; }
					if (c1 == '%') {
						state = STATE_TERM;
					} else {
						putchar(c);
						putchar(c1);
					}
				} else {
					putchar(c);
				}
				continue;
			}
			case STATE_TERM: {
				if (c == '%') {
					state = STATE_TERM_END;
				} else {
					if (i > 98) { return 1; }
					buffer[i] = c;
					i++;
				}
				continue;
			}
			case STATE_TERM_END: {
				if (c == '>') {
					state = STATE_TEXT;
					buffer[i] = '\0';
					if (i == t && strcmp(buffer, argv[1]) == 0) {
						printf("%s", filebuffer);
					}
				} else {
					return 1;
				}
				continue;
			}
		}
	}
	return 0;
}
