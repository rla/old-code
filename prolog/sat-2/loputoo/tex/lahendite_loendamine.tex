% Kasutatavad makrod
% Lemma väljastamine

\newcommand{\lemma}[1]{
	\hspace{1cm}\textbf{Lemma \arabic{lemma_counter}\stepcounter{lemma_counter}. \textit{#1}}
}

% Teoreemi väljastamine

\newcommand{\theorem}[1]{

	\arabic{theorem_counter}\stepcounter{theorem_counter}. \textit{#1}} }

\newcommand{\proof}[0]{

}
